\subsection{Code}
\begin{minted}{vhdl}
LIBRARY ieee;
USE ieee.std_logic_1164.ALL;

ENTITY Exc3 IS
	PORT (
		c : IN std_logic_vector(1 DOWNTO 0);
		HEX0 : OUT std_logic_vector(0 TO 6)
	);
END Exc3;

ARCHITECTURE behavior OF Exc3 IS
	SIGNAL HEX : std_logic_vector(0 TO 6);
BEGIN
	HEX0 <= NOT(HEX); -- 7SEGLED is active low.
	WITH c SELECT
	HEX <= "0111101" WHEN "00", 
	       "1001111" WHEN "01", 
	       "0110000" WHEN "10", 
	       "1111110" WHEN "11", 
	       "0000000" WHEN OTHERS;
END behavior;
\end{minted}
